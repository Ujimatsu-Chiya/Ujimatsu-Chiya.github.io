#include <bits/stdc++.h>
typedef long long ll;
using namespace std;
const int N=14;
const ll x0=N&1?pow(10,(N-1)>>1)*2:pow(10,(N-2)>>1)*7;
ll cnt=0;
void dfs(int f,ll v,ll l,ll r){
    cnt+=r-l+1;
    for(ll x=l,y=min(r,(l+v-1)/v*v);x<=r;x=y+1,y=min(x+v-1,r)){
        ll w=((x+v-1)/v+v)>>1;
        if(w==v) continue;
        dfs(f+1,w,x,y);
    }
}
int main(){
    ll l=pow(10,N-1),r=pow(10,N)-1;
    dfs(1,x0,l,r);
    double ans=1.0*cnt/(r-l+1);
    printf("%.10f\n",ans);
}
